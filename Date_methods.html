//SetMethods
            Method	Description
            setFullYear()	set year as a four digit number (yyyy)
            setMonth()	set month as a number (0-11)
            setDate()	set day as a number (1-31)
            setDay()	set weekday as a number (0-6)
            setHours()	set hour (0-23)
            setMinutes()	set minute (0-59)
            setSeconds()	set second (0-59)
            setMilliseconds()	set millisecond (0-999)
            setTime()	set time (milliseconds since January 1, 1970)
    
            //GetMethods
            Method	Description
            getFullYear()	Get year as a four digit number (yyyy)
            getMonth()	Get month as a number (0-11)
            getDate()	Get day as a number (1-31)
            getDay()	Get weekday as a number (0-6)
            getHours()	Get hour (0-23)
            getMinutes()	Get minute (0-59)
            getSeconds()	Get second (0-59)
            getMilliseconds()	Get millisecond (0-999)
            getTime()	Get time (milliseconds since January 1, 1970)
